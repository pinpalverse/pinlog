cmake_minimum_required(VERSION 4.3)
project(pinlog VERSION 1.0 DESCRIPTION "The logging library for pinpal" LANGUAGES C)
include(GNUInstallDirs)

add_library(pinlog SHARED src/pinlog.c)
add_library(pinlog_static STATIC src/pinlog.c)
set_target_properties(pinlog PROPERTIES
    VERSION ${PROJECT_VERSION}
    SOVERSION ${PROJECT_VERSION_MAJOR}
    PUBLIC_HEADER include/pinlog/pinlog.h)

set_target_properties(pinlog_static PROPERTIES
    OUTPUT_NAME pinlog 
    CLEAN_DIRECT_OUTPUT 1)

target_include_directories(pinlog 
    PRIVATE .
    INTERFACE
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>)


target_include_directories(pinlog_static
    PRIVATE .
    INTERFACE
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>)

configure_file(pinlog.pc.in pinlog.pc @ONLY)


add_executable(example example.c)
target_link_libraries(example pinlog)
target_include_directories(example PRIVATE include)

set_target_properties(example PROPERTIES
    INSTALL_RPATH "${CMAKE_INSTALL_PREFIX}/${CMAKE_INSTALL_LIBDIR}"
    INSTALL_RPATH_USE_LINK_PATH TRUE)


# Create simple config file for find_package
set(CONFIG_FILE_CONTENT "
# Generated by CMake for pinlog

set(PINLOG_FOUND TRUE)
set(PINLOG_VERSION ${PROJECT_VERSION})
set(PINLOG_LIBRARIES pinlog)
set(PINLOG_INCLUDE_DIRS ${CMAKE_INSTALL_FULL_INCLUDEDIR})

# Create imported target
add_library(pinlog::pinlog INTERFACE IMPORTED)
set_target_properties(pinlog::pinlog PROPERTIES
    INTERFACE_INCLUDE_DIRECTORIES \"${CMAKE_INSTALL_FULL_INCLUDEDIR}\"
    INTERFACE_LINK_LIBRARIES \"${CMAKE_INSTALL_FULL_LIBDIR}/libpinlog.so\"
)
")

file(WRITE ${CMAKE_BINARY_DIR}/pinlogConfig.cmake "${CONFIG_FILE_CONTENT}")

# Also create version file
set(VERSION_FILE_CONTENT "
set(PACKAGE_VERSION \"${PROJECT_VERSION}\")
if(NOT \"\${PACKAGE_VERSION}\" VERSION_EQUAL \"\${PACKAGE_FIND_VERSION}\")
    set(PACKAGE_VERSION_COMPATIBLE FALSE)
else()
    set(PACKAGE_VERSION_COMPATIBLE TRUE)
    if(\${PACKAGE_VERSION} VERSION_EQUAL \${PACKAGE_FIND_VERSION})
        set(PACKAGE_VERSION_EXACT TRUE)
    endif()
endif()
")
file(WRITE ${CMAKE_BINARY_DIR}/pinlogConfigVersion.cmake "${VERSION_FILE_CONTENT}")

# Install everything
install(TARGETS pinlog pinlog_static
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    PUBLIC_HEADER DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}/pinlog)

install(FILES 
    ${CMAKE_BINARY_DIR}/pinlog.pc
    DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/pkgconfig)

install(FILES 
    ${CMAKE_BINARY_DIR}/pinlogConfig.cmake
    ${CMAKE_BINARY_DIR}/pinlogConfigVersion.cmake
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/pinlog)
